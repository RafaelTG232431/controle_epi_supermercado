function verificarOutroAlmoxarifado() { const item = document.getElementById("item_almoxarifado").value; const campoOutro = document.getElementById("item_outro"); if (item === "Outro") { campoOutro.style.display = "block"; campoOutro.required = true; } else { campoOutro.style.display = "none"; campoOutro.required = false; campoOutro.value = ""; } }
